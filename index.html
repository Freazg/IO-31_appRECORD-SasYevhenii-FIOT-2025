1. Селектор тегу:

Опис: Селектор тегу застосовує стилі до всіх елементів вказаного типу на сторінці.

2. Селектор класу:

HTML:

<div class="container">
    <section class="hero">Контент</section>
</div>

CSS:

.container {
    max-width: 1200px;
    margin: 0 auto;
}

.hero {
    background: white;
    padding: 40px;
    border-radius: 15px;
}

Опис: Селектор класу (починається з крапки) дозволяє застосувати стилі до елементів з певним класом. Один клас може використовуватись багато разів на сторінці.

3. Селектор ідентифікатора:

HTML:

<img id="profile-photo" src="photo.jpg" alt="Фото">

CSS:

    width: 150px;
    height: 150px;
    border-radius: 50%;
    border: 5px solid white;
}

Опис: Селектор ID (починається з #) використовується для унікальних елементів. ID має бути унікальним на всій сторінці.

Breakpoints (точки перелому):

  • Desktop: 1200px і більше - повна версія сайту

ЗАВДАННЯ №1 - Перевірка чверті години

Умова: Збережіть введене значення часу (год:хвил) у змінну hour. Вивести у Console використовуючи шаблонний рядок. Перевірити до якої чверті години відноситься введене значення.

JavaScript код:

function checkTime() {
    let hour = prompt("Введіть час у форматі год:хвил:");
    if (hour === null || hour === "") {
        alert("Ви не ввели час!");
        return;
    }
    console.log(`Введений час: ${hour}`);
    let parts = hour.split(":");
    let minutes = parseInt(parts[1]);
    
    let quarter;
    if (minutes >= 0 && minutes < 15) {
        quarter = "Перша чверть години";
    } else if (minutes >= 15 && minutes < 30) {
        quarter = "Друга чверть години";
    } else if (minutes >= 30 && minutes < 45) {
        quarter = "Третя чверть години";
    } else {
        quarter = "Четверта чверть години";
    }
    alert(quarter);
}

}

function makeArray(firstArray, secondArray, maxLength) {
    let newArray = firstArray.concat(secondArray);

ЗАВДАННЯ №7 - Робота з масивом (Варіант 10)

Умова: Згенерувати масив, знайти мінімальний елемент, перемістити на початок, відсортувати методом вибору.

JavaScript код:

function findMinIndex(array) {
    let minIndex = 0;
    for (let i = 1; i < array.length; i++) {
        if (array[i] < array[minIndex]) minIndex = i;
    }
    return minIndex;
}

function moveMinToStart(array) {
    let minIndex = findMinIndex(array);
    let minValue = array[minIndex];
    for (let i = minIndex; i > 0; i--) {
        array[i] = array[i - 1];
    }
    array[0] = minValue;
    return array;
}

function selectionSortDesc(array) {
    for (let i = 0; i < array.length - 1; i++) {
        let maxIndex = i;
        for (let j = i + 1; j < array.length; j++) {
            if (array[j] > array[maxIndex]) maxIndex = j;
        }
        if (maxIndex !== i) {
            [array[i], array[maxIndex]] = [array[maxIndex], array[i]];
        }
    }
    return array;

    let array = [];
    for (let i = 0; i < rows; i++) {
        let row = [];

            row.push(Math.floor(Math.random() * 61) - 30);

    return array;
}
